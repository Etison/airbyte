82def records_with_state(records, state, stream_mapping, state_cursor_paths) -> Iterable[Tuple[Any, Any]]: 

83 """Iterate over records and return cursor value with corresponding cursor value from state""" 

84 for record in records: 

85 stream_name = record.record.stream 

86 stream = stream_mapping[stream_name] 

87 helper = JsonSchemaHelper(schema=stream.stream.json_schema) 

88 cursor_field = helper.field(stream.cursor_field) 

89 record_value = cursor_field.parse(record=record.record.data) 

90 try: 

91 # first attempt to parse the state value assuming the state object is namespaced on stream names 

92 state_value = cursor_field.parse(record=state[stream_name], path=state_cursor_paths[stream_name]) 

93 except KeyError: 

94 # try second time as an absolute path in state file (i.e. bookmarks -> stream_name -> column -> value) 

95 state_value = cursor_field.parse(record=state, path=state_cursor_paths[stream_name]) 

96 yield record_value, state_value

102 stream_mapping = {stream.stream.name: stream for stream in configured_catalog_for_incremental.streams} 

103 

104 output = docker_runner.call_read(connector_config, configured_catalog_for_incremental) 

105 records_1 = filter_output(output, type_=Type.RECORD) 

106 states_1 = filter_output(output, type_=Type.STATE) 

107 

108 assert states_1, "Should produce at least one state" 

109 assert records_1, "Should produce at least one record" 

110 

111 latest_state = states_1[-1].state.data 

112 for record_value, state_value in records_with_state(records_1, latest_state, stream_mapping, cursor_paths): 

113 assert ( 

114 record_value <= state_value 

115 ), "First incremental sync should produce records younger or equal to cursor value from the state" 

116 

117 output = docker_runner.call_read_with_state(connector_config, configured_catalog_for_incremental, state=latest_state) 

118 records_2 = filter_output(output, type_=Type.RECORD) 

119 

120 for record_value, state_value in records_with_state(records_2, latest_state, stream_mapping, cursor_paths): 

121 assert ( 

122 record_value >= state_value 

123 ), "Second incremental sync should produce records older or equal to cursor value from the state"
